# 사용한 include들

```c++
#include <iostream> /// 입출력을 위함
#include <string> // string 변수를 만들기위함
#include "monster.h" // 몬스터 클래스
#include "slime.h" // 슬라임(몬스터의 파생)클래스
#include "player.h" // 플레이어 클래스
#include "warrior.h" // 전사(플레이어의 파생)클래스
#include "magician.h" // 법사(플레이어의 파생)클래스
#include "thief.h" // 도적(플레이어의 파생)클래스
#include "archer.h" // 궁수(플레이어의 파생)클래스
#include <thread> // 출력 시간관련 
#include <chrono> // 출력 시간관련
```
